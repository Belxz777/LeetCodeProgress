# Задачка на нахождения топ k самых встречающихся чисел в массиве
## Приведенны два решение простое и бакет сорт 
Код: коммы есть
```
function topKFrequent(nums: number[], k: number): number[] {
    let map = new Map()
    for(const num of nums){
        map.set(num,(map.get(num) || 0) + 1)
    }
    // самое простое нативное решение
    // let s = Array.from(map.entries())
    // .sort((a,b)=> b[1] - a[1])
    // .slice(0,k)
    // .map( entry=> Number(entry[0]))
    // return s 
 
    // сортировка bucket sort
    let buckets = Array.from({length:nums.length +1},()=>[])
    // создаем [] количество раз столько сколько в nums элементов
    for(let [num,freq] of map.entries()){
        buckets[freq].push(num)
    }
    // записали значения  по индексу частоты появления
    let result = [];
    for(let i = buckets.length - 1;i>=0 && result.length < k;--i){
        for(let num of buckets[i]){
             result.push(num)
             if(result.length === k) break;
        }
       // тупо заносим с result самые последние цифры в из buckets || пока элементов в
       // массиве result не будет k
    }
    return result

    

};
```
